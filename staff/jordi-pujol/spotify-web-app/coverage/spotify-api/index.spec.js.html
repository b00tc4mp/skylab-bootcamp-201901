            const query = 'madonna'

                    expect(artists).to.exist
                    expect(artists instanceof Array).to.be.true
                    expect(artists.length).not.to.equal(0)
 
                    artists.forEach(({ name }) => expect(name.toLowerCase()).to.contain(query))
                })

        })
    })
